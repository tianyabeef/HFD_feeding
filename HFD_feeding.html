    others_ind <- order(-x)[-(1:top)]
    others <- sum(x[others_ind])
    x[others_ind] <- 0
    x <- c(x, others = others)
    x
  }
  tabletmp <- apply(table, 2, merge_low_abundance, vector_name = rownames(table))
